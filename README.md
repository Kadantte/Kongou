## Examples

```js
const kongou = require("kongou");
await kongou
  .get(223435)
  .then((data) => console.log(data))
  .catch((error) => console.log(error));
```

or

```js
const kongou = require("kongou");
await kongou
  .query("艦隊これくしょん")
  .then((data) => console.log(data))
  .catch((error) => console.log(error));
```
